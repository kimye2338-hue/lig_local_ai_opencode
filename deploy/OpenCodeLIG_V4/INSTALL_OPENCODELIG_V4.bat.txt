@echo off
rem ==========================================================================
rem INSTALL_OPENCODELIG_V4.bat - one-time install, re-runnable for updates.
rem HOW TO USE: rename this file from .bat.txt to .bat, then double-click.
rem It copies everything into %USERPROFILE%\OpenCodeLIG, strips the .txt
rem suffixes, backs up existing AGENTS files, and applies the config.
rem It never overwrites memory/MEMORY.md or checkpoints on update.
rem ASCII only, no PowerShell, no downloads.
rem ==========================================================================
setlocal
chcp 65001 >nul
set "SRC=%~dp0"
set "LIG=%USERPROFILE%\OpenCodeLIG"
set "UD=%USERPROFILE%\OpenCodeLIG_USERDATA"

echo [1/6] creating directories...
for %%D in ("%LIG%" "%LIG%\bin" "%LIG%\scripts" "%LIG%\proxy" "%LIG%\workspace" "%LIG%\docs" "%LIG%\logs" "%LIG%\secrets" "%LIG%\workspace\memory" "%LIG%\workspace\checkpoints" "%LIG%\workspace\skills" "%UD%" "%UD%\opencode_config" "%UD%\opencode_data" "%UD%\opencode_state" "%UD%\opencode_cache" "%UD%\diagnostics" "%UD%\backups" "%UD%\state") do if not exist "%%~D" mkdir "%%~D"

echo [2/6] backing up existing workspace policy files...
if not exist "%LIG%\workspace\AGENTS.md" goto SKIPBK
set "BK=%UD%\backups\agents_backup_%RANDOM%%RANDOM%"
mkdir "%BK%" >nul 2>nul
copy /y "%LIG%\workspace\AGENTS*.md" "%BK%\" >nul
echo        backup: %BK%
:SKIPBK

echo [3/6] copying files...
copy /y "%SRC%RUN_OPENCODE_LIG.bat.txt" "%LIG%\" >nul
copy /y "%SRC%RUN_OPENCODE_LIG_SAFE_PURE.bat.txt" "%LIG%\" >nul
copy /y "%SRC%DIAG_OPENCODE_LIG.bat.txt" "%LIG%\" >nul
copy /y "%SRC%APPLY_OR_REPAIR_OPENCODE_LIG.bat.txt" "%LIG%\" >nul
copy /y "%SRC%scripts\*.py.txt" "%LIG%\scripts\" >nul
copy /y "%SRC%proxy\*.py.txt" "%LIG%\proxy\" >nul
xcopy /e /y /q "%SRC%workspace_seed\*" "%LIG%\workspace\" >nul
copy /y "%SRC%RUNBOOK.md" "%LIG%\docs\" >nul
copy /y "%SRC%ARCHITECTURE_REFORM_PLAN.md" "%LIG%\docs\" >nul 2>nul
copy /y "%SRC%FEEDBACK_AND_WORK_PLAN.md" "%LIG%\docs\" >nul 2>nul

rem seed durable memory only if it does not exist yet (never overwrite)
if not exist "%LIG%\workspace\memory\MEMORY.md" copy /y "%LIG%\workspace\memory\MEMORY_SEED.md" "%LIG%\workspace\memory\MEMORY.md" >nul

echo [4/6] activating scripts (removing .txt suffix)...
for %%F in ("%LIG%\*.bat.txt") do call :STRIP "%%~fF"
for %%F in ("%LIG%\scripts\*.py.txt") do call :STRIP "%%~fF"
for %%F in ("%LIG%\proxy\*.py.txt") do call :STRIP "%%~fF"

echo [5/6] applying config...
where python >nul 2>nul
if errorlevel 1 (
  echo [WARN] Python not found in PATH - config not applied yet.
  echo        Fix PATH, then run %LIG%\APPLY_OR_REPAIR_OPENCODE_LIG.bat
) else (
  python "%LIG%\scripts\lig_apply_config.py"
)

echo [6/6] install complete.
echo.
echo NEXT STEPS:
echo   1. put opencode.exe into %LIG%\bin\   (from the offline package)
echo   2. run SET_LIG_SECRET.bat once        (provided separately, NOT in git)
echo   3. run %LIG%\DIAG_OPENCODE_LIG.bat    (expect GO)
echo   4. daily use: %LIG%\RUN_OPENCODE_LIG.bat
echo.
if not "%LIG_NONINTERACTIVE%"=="1" pause
exit /b 0

:STRIP
set "D=%~dp1"
set "N=%~n1"
if exist "%D%%N%" del /q "%D%%N%"
ren "%~1" "%N%"
exit /b
