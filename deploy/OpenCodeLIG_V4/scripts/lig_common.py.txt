def ensure_dirs():
    for d in [WS, BIN, LOGS, ROOT / "proxy", ROOT / "scripts", ROOT / "docs",
              ROOT / "secrets", CFGDIR, DIAGDIR, BACKUPS, STATEDIR,
              USERDATA / "opencode_data", USERDATA / "opencode_state",
              USERDATA / "opencode_cache",
              WS / "memory", WS / "checkpoints", WS / "skills"]:
        d.mkdir(parents=True, exist_ok=True)


# ---------------------------------------------------------------- secrets
def load_secrets():
    """Load KEY=VALUE lines from lig_local.env into os.environ (no overwrite)."""
    loaded = []
    try:
        for line in SECRET_FILE.read_text(encoding="utf-8", errors="replace").splitlines():
            line = line.strip()
            if not line or line.startswith("#") or "=" not in line:
                continue
            k, v = line.split("=", 1)
            k, v = k.strip(), v.strip()
            if k and k not in os.environ:
                os.environ[k] = v
                loaded.append(k)
    except OSError:
        pass
    return loaded


def secrets_ok():
    return bool(os.environ.get("LIG_PROXY_CHAT_UPSTREAM")) and bool(os.environ.get("LIG_API_KEY"))


def secrets_help():

            "  Fix    : run SET_LIG_SECRET.bat once (provided separately, NOT in git).\n"
            "  It writes: %s" % (
                ", ".join(k for k in ("LIG_PROXY_CHAT_UPSTREAM", "LIG_API_KEY")
                          if not os.environ.get(k)) or "-",
                SECRET_FILE))


# ---------------------------------------------------------------- environment
def build_env(registry_failfast=True):
    """One environment for launcher, diagnostics and headless runs alike."""
    env = os.environ.copy()
    env.update({
        "PYTHONUTF8": "1",
        "PYTHONIOENCODING": "utf-8",
        "OPENCODE_DISABLE_MODELS_FETCH": "1",
        "OPENCODE_CONFIG": str(CFG),
        "OPENCODE_USERDATA": str(USERDATA),
        "XDG_CONFIG_HOME": str(CFGDIR),
        "XDG_DATA_HOME": str(USERDATA / "opencode_data"),
        "XDG_STATE_HOME": str(USERDATA / "opencode_state"),
        "XDG_CACHE_HOME": str(USERDATA / "opencode_cache"),
        "AGENTOPS_HOME": str(WS),
        # keep localhost traffic away from any corporate proxy settings
        "NO_PROXY": "127.0.0.1,localhost",
        # proxy runtime settings (secrets already merged into os.environ)
        "LIG_PROXY_HOST": HOST,
        "LIG_PROXY_PORT": PORT,
        "LIG_LOG_DIR": str(LOGS),
        "LIG_PROXY_MODEL": MODEL_ID,
        "LIG_PROXY_TEMPERATURE": os.environ.get("LIG_PROXY_TEMPERATURE", "0"),
        "LIG_PROXY_TOP_P": os.environ.get("LIG_PROXY_TOP_P", "1"),
    })
    if registry_failfast:
        env["NPM_CONFIG_REGISTRY"] = FAILFAST_REGISTRY
    else:
        env.pop("NPM_CONFIG_REGISTRY", None)
    return env


# ---------------------------------------------------------------- mode state
def get_mode():
    try:
        first = MODE_FILE.read_text(encoding="utf-8").splitlines()[0].strip().lower()
        if first in ("normal", "pure"):
            return first
    except (OSError, IndexError):
        pass
    return "pure"  # safe default until a diagnostic proves normal is fast


def set_mode(mode, reason=""):
    STATEDIR.mkdir(parents=True, exist_ok=True)
    MODE_FILE.write_text("%s\n# set %s\n# reason: %s\n" % (
        mode, datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S"), reason),
        encoding="utf-8")

def write_config(backup=True):
    ensure_dirs()
    if backup and CFG.exists():
        stamp = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
        (BACKUPS / ("opencode.proxy.%s.json" % stamp)).write_text(
            CFG.read_text(encoding="utf-8", errors="replace"), encoding="utf-8")
    cfg = build_config()
    CFG.write_text(json.dumps(cfg, ensure_ascii=False, indent=2) + "\n", encoding="utf-8")
    return cfg

# ---------------------------------------------------------------- proxy
def proxy_health(timeout=3):
    try:
        with urllib.request.urlopen("http://%s:%s/health" % (HOST, PORT), timeout=timeout) as r:
            body = r.read(500).decode("utf-8", "replace")
            identity = "unknown"
            try:
                identity = json.loads(body).get("proxy", "unknown")
            except ValueError:
                pass
            return r.status == 200, identity, body
    except Exception as exc:
        return False, "unreachable", repr(exc)


def start_proxy(env=None, wait=30):
    """Start the proxy if not healthy. Returns (ok, identity)."""
    ok, identity, _ = proxy_health()
    if ok:
        return True, identity
    if not PROXY_PY.exists():
        return False, "proxy script missing: %s" % PROXY_PY
    env = env or build_env()
    flags = getattr(subprocess, "CREATE_NEW_CONSOLE", 0) if os.name == "nt" else 0
    subprocess.Popen([sys.executable, str(PROXY_PY)], cwd=str(WS), env=env,
                     creationflags=flags)
    for _ in range(wait):
        time.sleep(1)
        ok, identity, _ = proxy_health()
        if ok:
            return True, identity
    return False, identity


# ---------------------------------------------------------------- subprocess
def _to_text(v):
    if v is None:
        return ""
    if isinstance(v, bytes):
        return v.decode("utf-8", "replace")
    return v


def run_capture(args, cwd=None, env=None, timeout=60):
    """UTF-8/cp949-safe capture. Never raises on timeout; returns a dict."""
    t0 = time.time()
    try:
        cp = subprocess.run([str(a) for a in args], cwd=str(cwd) if cwd else None,
                            env=env, capture_output=True, text=True,
                            encoding="utf-8", errors="replace", timeout=timeout)
        return {"seconds": round(time.time() - t0, 1), "returncode": cp.returncode,
                "timeout": False, "stdout": cp.stdout or "", "stderr": cp.stderr or ""}
    except subprocess.TimeoutExpired as e:
        return {"seconds": round(time.time() - t0, 1), "returncode": None,
                "timeout": True, "stdout": _to_text(e.stdout), "stderr": _to_text(e.stderr)}
    except OSError as e:
        return {"seconds": round(time.time() - t0, 1), "returncode": -1,
                "timeout": False, "stdout": "", "stderr": repr(e)}


def tail(text, lines=40):
    return "\n".join((text or "").splitlines()[-lines:])


def tail_file(path, lines=40):
    try:
        return tail(pathlib.Path(path).read_text(encoding="utf-8", errors="replace"), lines)
    except OSError as e:
        return "(unreadable: %r)" % e


def recent_logs(limit=3, max_bytes=5_000_000):
    """Newest small *.log files under USERDATA and LOGS (for failure reports)."""
    found = []
    for base in (USERDATA, LOGS):
        if base.exists():
            for p in base.rglob("*.log"):
                try:
                    st = p.stat()
                    if st.st_size <= max_bytes:
                        found.append((st.st_mtime, p))
                except OSError:
                    pass
    found.sort(reverse=True)
    return [p for _, p in found[:limit]]
