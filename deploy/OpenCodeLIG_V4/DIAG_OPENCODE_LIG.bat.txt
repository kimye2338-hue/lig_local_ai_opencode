@echo off
rem ==========================================================================
rem DIAG_OPENCODE_LIG.bat (V4) - the ONE diagnostic. Prints a GO/NO-GO block
rem and a report path. Safe to run any time; restores plugins afterwards.
rem Set LIG_NONINTERACTIVE=1 to skip the final pause (automation).

setlocal
chcp 65001 >nul
set "PYTHONUTF8=1"
set "PYTHONIOENCODING=utf-8"
where python >nul 2>nul || (echo [ERROR] Python not found in PATH.& pause & exit /b 1)
python "%USERPROFILE%\OpenCodeLIG\scripts\lig_diag.py"
set "RC=%ERRORLEVEL%"
echo.
if not "%LIG_NONINTERACTIVE%"=="1" pause
exit /b %RC%
