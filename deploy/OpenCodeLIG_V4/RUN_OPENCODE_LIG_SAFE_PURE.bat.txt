@echo off
rem ==========================================================================
rem RUN_OPENCODE_LIG_SAFE_PURE.bat (V4) - explicit fallback launcher.
rem Forces --pure (no local plugins) regardless of saved startup mode.
rem Use when normal startup is slow. File-write guard still active (it lives
rem in the local proxy), destructive bash still denied by config permissions.
rem ==========================================================================
if "%~1"=="__MAX" goto MAIN
start "OpenCodeLIG (PURE)" /max cmd /c "%~f0" __MAX
exit /b

:MAIN
chcp 65001 >nul
set "PYTHONUTF8=1"
set "PYTHONIOENCODING=utf-8"
where python >nul 2>nul || (echo [ERROR] Python not found in PATH.& pause & exit /b 1)
python "%USERPROFILE%\OpenCodeLIG\scripts\lig_launch.py" --pure
set "RC=%ERRORLEVEL%"
if not "%RC%"=="0" (
  echo.
  echo [EXIT] code=%RC%   ^(10 = secrets missing: run SET_LIG_SECRET.bat once^)
  pause
)
exit /b %RC%
