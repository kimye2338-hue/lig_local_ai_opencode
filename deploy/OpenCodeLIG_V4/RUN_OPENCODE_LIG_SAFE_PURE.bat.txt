@echo off

rem RUN_OPENCODE_LIG_SAFE_PURE.bat (V4) - explicit fallback launcher.
rem Forces safe mode (--pure) regardless of saved startup mode. No feature

  echo [종료코드 %RC%] 화면의 [문제]/[해결] 안내를 따라 주세요.
  pause
)
