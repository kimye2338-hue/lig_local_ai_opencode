# -*- coding: utf-8 -*-
from __future__ import annotations
import json, shutil, subprocess, sys, os
from datetime import datetime
from pathlib import Path

MARKER = '<!-- OPENCODE_AGENTOPS_V3_1_COGROWTH -->'
END_MARKER = '<!-- /OPENCODE_AGENTOPS_V3_1_COGROWTH -->'
PATCH_VERSION = "opencode-agentops-v3.1-cogrowth"

# Files that MUST be present in the payload and after install. The command-guard
# plugin is a P0 safety control, so the installer fails loudly if it is missing
# instead of silently shipping a build without the corrupted-command guard.
REQUIRED_PAYLOAD_FILES = [
    "agent_ops/agentops.py",
    "agent_ops/command_guard.py",
    ".opencode/plugins/command-guard.ts",
    ".opencode/plugins/compaction-handoff.ts",
]
# Top-level payload entries that must never be copied into the target project.
PAYLOAD_SKIP_TOP = {"installers_light", ".agentops_backup", ".git"}

def read_text(path: Path) -> str:
    if path.exists():
        return path.read_text(encoding="utf-8", errors="replace")
    return ""

def write_text(path: Path, text: str) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(str(text).replace("\r\n", "\n"), encoding="utf-8")

def write_json(path: Path, data) -> None:
    write_text(path, json.dumps(data, ensure_ascii=False, indent=2))

def backup(root: Path, path: Path, backup_root: Path) -> None:
    if not path.exists():
        return
    try:
        rel = path.relative_to(root)
    except Exception:
        rel = Path(path.name)
    dst = backup_root / rel
    dst.parent.mkdir(parents=True, exist_ok=True)
    if path.is_dir():
        if dst.exists():
            shutil.rmtree(dst)
        shutil.copytree(path, dst)
    else:
        shutil.copy2(path, dst)

def merge_unique(existing, additions):
    out = []
    if isinstance(existing, list):
        for x in existing:
            if isinstance(x, str) and not x.startswith(".agent-memory") and x not in out:
                out.append(x)
    for x in additions:
        if x not in out:
            out.append(x)
    return out

def merge_opencode(root: Path, backup_root: Path):
    p = root / "opencode.json"
    backup(root, p, backup_root)
    try:
        cfg = json.loads(read_text(p)) if p.exists() else {}
        if not isinstance(cfg, dict):
            cfg = {}
    except Exception:
        backup(root, p, backup_root / "invalid")
        cfg = {}
    cfg.setdefault("$schema", "https://opencode.ai/config.json")
    cfg["instructions"] = merge_unique(cfg.get("instructions"), ["AGENTS.md", "agent_ops/AGENTOPS_RULES.md"])
    comp = cfg.get("compaction") if isinstance(cfg.get("compaction"), dict) else {}
    comp.setdefault("auto", True)
    comp.setdefault("prune", True)
    comp.setdefault("reserved", 8000)
    cfg["compaction"] = comp
    watcher = cfg.get("watcher") if isinstance(cfg.get("watcher"), dict) else {}
    watcher["ignore"] = merge_unique(watcher.get("ignore"), ["agent_ops/logs/**", "agent_ops/archive/**", "agent_ops/locks/**", "portal_research/screenshots/**", "portal_research/html_snapshots/**", ".agentops_backup/**", "**/*.bak", "**/*.backup"])
    cfg["watcher"] = watcher
    perm = cfg.get("permission") if isinstance(cfg.get("permission"), dict) else {}
    for k, v in {"read":"allow","grep":"allow","glob":"allow","todowrite":"allow","edit":"ask","bash":"ask","question":"deny","webfetch":"deny","websearch":"deny","external_directory":"deny"}.items():
        perm.setdefault(k, v)
    cfg["permission"] = perm
    if isinstance(cfg.get("agent"), dict):
        cfg["agent"].pop("agentops-supervisor", None)
    write_json(p, cfg)

def update_agents_md(root: Path):
    p = root / "AGENTS.md"
    existing = read_text(p)
    block = f"""
{MARKER}
# OpenCode AgentOps v3.1 Co-Growth

Before starting:
1. Read `agent_ops/AGENTOPS_RULES.md`.
2. Read `agent_ops/state/RESUME_PLAN.md`.
3. Read `agent_ops/state/COMPACT_HANDOFF.md`.
4. Use `/continue` for one bounded task only.
5. Do not run long loops inside OpenCode bash.
6. External continuous execution belongs to `RUN_AGENTOPS_ORCHESTRATOR.bat`.

Memory:
- Do not load growing memory files into instructions.
- Use `.agent-memory/memory.jsonl` as source of truth.
- Use `/remember` when the user corrects behavior.

Safety:
- Queue items and next steps are planned, not approved.
- Risky actions need explicit current-session user approval.
{END_MARKER}
""".strip()
    if MARKER in existing and END_MARKER in existing:
        start = existing.index(MARKER)
        end = existing.index(END_MARKER) + len(END_MARKER)
        existing = existing[:start].rstrip() + "\n\n" + block + "\n" + existing[end:].lstrip()
    else:
        existing = existing.rstrip() + "\n\n" + block + "\n"
    write_text(p, existing)

def find_payload(here: Path, root: Path):
    # Probe the historical name AND the current package layout. In this package the
    # editable source IS the installer's parent (installers_light/ lives inside the
    # source tree), and the Opus-review reference zips ship it as current_source/.
    candidates = [
        here / "agentops_v3_1_payload",
        root / "agentops_v3_1_payload",
        here / "current_source",
        root / "current_source",
        here.parent / "agentops_v3_1_payload",
        here.parent / "current_source",
        here.parent,  # PROJECT_FULL_SOURCE_TO_EDIT / current_source itself
    ]
    seen = set()
    for cand in candidates:
        try:
            key = str(cand.resolve())
        except Exception:
            key = str(cand)
        if key in seen:
            continue
        seen.add(key)
        try:
            if cand.exists() and (cand / "agent_ops" / "agentops.py").is_file():
                return cand
        except Exception:
            continue
    return None

def iter_payload_files(payload: Path):
    for src in sorted(payload.rglob("*")):
        if not src.is_file():
            continue
        rel = src.relative_to(payload)
        if rel.parts and rel.parts[0] in PAYLOAD_SKIP_TOP:
            continue
        if any(part == "__pycache__" for part in rel.parts):
            continue
        yield src, rel

def main():
    argv = sys.argv[1:]
    dry_run = "--dry-run" in argv
    root = Path.cwd()
    here = Path(__file__).resolve().parent
    payload = find_payload(here, root)
    if payload is None:
        print("[ERROR] AgentOps payload not found.")
        print("        Looked for agentops_v3_1_payload/ or current_source/ (or the")
        print("        installer's parent source tree) containing agent_ops/agentops.py.")
        return 1
    print("[INFO] Using payload:", payload)

    # Validate required files exist in the payload BEFORE touching the project.
    missing_payload = [f for f in REQUIRED_PAYLOAD_FILES if not (payload / f).is_file()]
    if missing_payload:
        print("[ERROR] payload is missing required files:")
        for f in missing_payload:
            print("        -", f)
        return 1

    files = list(iter_payload_files(payload))
    guard_rel = ".opencode/plugins/command-guard.ts"
    will_copy_guard = any(rel.as_posix() == guard_rel for _, rel in files)

    if dry_run:
        print("[DRY-RUN] No files will be written.")
        print("[DRY-RUN] Payload files to copy:", len(files))
        print("[DRY-RUN] command-guard.ts present in payload and will be copied:", will_copy_guard)
        print("[DRY-RUN] Required-files check: OK")
        for f in REQUIRED_PAYLOAD_FILES:
            print("          -", f)
        if not will_copy_guard:
            print("[ERROR] command-guard.ts would NOT be copied")
            return 1
        print("DRY_RUN_OK")
        return 0

    backup_root = root / ".agentops_backup" / datetime.now().strftime("%Y%m%d_%H%M%S")
    backup(root, root / "opencode.json", backup_root)
    backup(root, root / "AGENTS.md", backup_root)
    merge_opencode(root, backup_root)
    copied = 0
    for src, rel in files:
        dst = root / rel
        if str(rel).startswith(".agent-memory") and dst.exists():
            continue
        try:
            if src.resolve() == dst.resolve():
                continue  # payload already in place (installing in-tree)
        except Exception:
            pass
        backup(root, dst, backup_root)
        dst.parent.mkdir(parents=True, exist_ok=True)
        shutil.copy2(src, dst)
        copied += 1
    update_agents_md(root)

    # Hard guarantee: required files (incl. the command-guard plugin) exist now.
    missing_after = [f for f in REQUIRED_PAYLOAD_FILES if not (root / f).is_file()]
    if missing_after:
        print("[ERROR] required files missing after install:")
        for f in missing_after:
            print("        -", f)
        return 1

    subprocess.run([sys.executable, str(root / "agent_ops" / "agentops.py"), "init"], cwd=str(root), check=False)
    result = subprocess.run([sys.executable, str(root / "agent_ops" / "agentops.py"), "verify"], cwd=str(root), check=False)
    print("INSTALL_OK")
    print("Patch:", PATCH_VERSION)
    print("Project root:", root)
    print("Files copied:", copied)
    print("Command guard plugin installed:", (root / guard_rel).is_file())
    print("Next: ocode -> /doctor -> /resume -> /continue")
    if result.returncode != 0:
        print("[WARN] verify returned non-zero. Check agent_ops/reports/VERIFICATION_REPORT.md")
    return 0

if __name__ == "__main__":
    raise SystemExit(main())
