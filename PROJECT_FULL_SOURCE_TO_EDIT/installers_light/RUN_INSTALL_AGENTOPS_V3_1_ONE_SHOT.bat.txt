@echo off
setlocal EnableExtensions
set PYTHONUTF8=1
set PYTHONIOENCODING=utf-8

set "SCRIPT=%~dp0INSTALL_OPENCODE_AGENTOPS_V3_1_COGROWTH.py.txt"
if not exist "%SCRIPT%" set "SCRIPT=%CD%\INSTALL_OPENCODE_AGENTOPS_V3_1_COGROWTH.py.txt"
set "LOG=%CD%\agentops_v31_install_log.txt"

echo.
echo OpenCode AgentOps v3.1 safe installer
echo Script: %SCRIPT%
echo Log: %LOG%
echo.

if not exist "%SCRIPT%" (
  echo [ERROR] installer script not found.
  echo Put this BAT in the same folder as INSTALL_OPENCODE_AGENTOPS_V3_1_COGROWTH.py.txt
  echo and agentops_v3_1_payload folder.
  pause
  exit /b 1
)

echo [INFO] Checking Python candidates... > "%LOG%"
echo SCRIPT=%SCRIPT% >> "%LOG%"
echo CD=%CD% >> "%LOG%"
echo. >> "%LOG%"

where py >> "%LOG%" 2>&1
if not errorlevel 1 (
  echo [TRY] py -3.11
  echo [TRY] py -3.11 >> "%LOG%"
  py -3.11 "%SCRIPT%" >> "%LOG%" 2>&1
  if not errorlevel 1 goto SUCCESS

  echo [TRY] py -3
  echo [TRY] py -3 >> "%LOG%"
  py -3 "%SCRIPT%" >> "%LOG%" 2>&1
  if not errorlevel 1 goto SUCCESS
)

where python >> "%LOG%" 2>&1
if not errorlevel 1 (
  echo [TRY] python
  echo [TRY] python >> "%LOG%"
  python "%SCRIPT%" >> "%LOG%" 2>&1
  if not errorlevel 1 goto SUCCESS
)

where python3 >> "%LOG%" 2>&1
if not errorlevel 1 (
  echo [TRY] python3
  echo [TRY] python3 >> "%LOG%"
  python3 "%SCRIPT%" >> "%LOG%" 2>&1
  if not errorlevel 1 goto SUCCESS
)

if exist "%LOCALAPPDATA%\Programs\Python\Python311\python.exe" (
  echo [TRY] LOCALAPPDATA Python311
  echo [TRY] LOCALAPPDATA Python311 >> "%LOG%"
  "%LOCALAPPDATA%\Programs\Python\Python311\python.exe" "%SCRIPT%" >> "%LOG%" 2>&1
  if not errorlevel 1 goto SUCCESS
)

if exist "%USERPROFILE%\AppData\Local\Programs\Python\Python311\python.exe" (
  echo [TRY] USERPROFILE Python311
  echo [TRY] USERPROFILE Python311 >> "%LOG%"
  "%USERPROFILE%\AppData\Local\Programs\Python\Python311\python.exe" "%SCRIPT%" >> "%LOG%" 2>&1
  if not errorlevel 1 goto SUCCESS
)

if exist "C:\Python311\python.exe" (
  echo [TRY] C:\Python311
  echo [TRY] C:\Python311 >> "%LOG%"
  "C:\Python311\python.exe" "%SCRIPT%" >> "%LOG%" 2>&1
  if not errorlevel 1 goto SUCCESS
)

echo.
echo [ERROR] Python failed or installer failed.
echo See log:
echo   %LOG%
echo.
echo Last log lines:
echo ----------------------------------------
powershell -NoProfile -Command "Get-Content -Path '%LOG%' -Tail 80" 2>nul
echo ----------------------------------------
echo.
echo Manual checks:
echo   py -0p
echo   where python
echo   python --version
echo   py -3.11 "%SCRIPT%"
echo.
pause
exit /b 1

:SUCCESS
echo.
echo [OK] Install completed.
echo See log:
echo   %LOG%
echo.
echo Next:
echo   ocode
echo   /doctor
echo   /resume
echo   /continue
echo.
