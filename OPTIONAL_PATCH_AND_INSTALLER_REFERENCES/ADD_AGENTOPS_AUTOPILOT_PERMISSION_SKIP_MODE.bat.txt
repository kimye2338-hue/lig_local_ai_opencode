@echo off
setlocal EnableExtensions
set PYTHONUTF8=1
set PYTHONIOENCODING=utf-8

set "SELF=%~f0"
set "ROOT=%CD%"
set "WORK=%TEMP%\agentops_autopilot_%RANDOM%%RANDOM%"
set "PYFILE=%WORK%\add_agentops_autopilot.py"
set "LOG=%ROOT%\agentops_autopilot_patch_log.txt"

echo.
echo Installing AgentOps Autopilot permission-skip mode...
echo Target: %ROOT%
echo Log: %LOG%
echo.

if not exist "%WORK%" mkdir "%WORK%"

set "AGENTOPS_AUTOPILOT_SELF=%SELF%"
set "AGENTOPS_AUTOPILOT_ROOT=%ROOT%"
set "AGENTOPS_AUTOPILOT_LOG=%LOG%"

> "%PYFILE%" echo import base64, json, os, pathlib, datetime
>> "%PYFILE%" echo marker = "__AGENTOPS_AUTOPILOT_PAYLOAD_BASE64__"
>> "%PYFILE%" echo self_path = pathlib.Path(os.environ["AGENTOPS_AUTOPILOT_SELF"])
>> "%PYFILE%" echo root = pathlib.Path(os.environ["AGENTOPS_AUTOPILOT_ROOT"]).resolve()
>> "%PYFILE%" echo log = pathlib.Path(os.environ["AGENTOPS_AUTOPILOT_LOG"]).resolve()
>> "%PYFILE%" echo text = self_path.read_text(encoding="ascii", errors="ignore")
>> "%PYFILE%" echo if marker not in text: raise SystemExit("payload marker not found")
>> "%PYFILE%" echo tail = text.rsplit(marker, 1)[1]
>> "%PYFILE%" echo allowed = set("ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/=")
>> "%PYFILE%" echo payload_b64 = "".join(ch for ch in tail if ch in allowed)
>> "%PYFILE%" echo data = json.loads(base64.b64decode(payload_b64).decode("utf-8"))
>> "%PYFILE%" echo for rel, content in data.items():
>> "%PYFILE%" echo     path = root / rel
>> "%PYFILE%" echo     path.parent.mkdir(parents=True, exist_ok=True)
>> "%PYFILE%" echo     path.write_text(content.replace("\\r\\n", "\\n"), encoding="utf-8")
>> "%PYFILE%" echo ag = root / "AGENTS.md"
>> "%PYFILE%" echo block = "\\n\\n<!-- AGENTOPS_AUTOPILOT_MODE -->\\n# AgentOps Autopilot Mode\\n\\nUse `agentops-supervisor` for cautious planning/review. Use `agentops-autopilot` or `/autopilot <task>` when project-local permission prompts should be skipped. Autopilot allows project-local edit/bash but still denies external_directory, webfetch, and websearch. Risky portal actions remain blocked unless explicitly approved in the current session.\\n<!-- /AGENTOPS_AUTOPILOT_MODE -->\\n"
>> "%PYFILE%" echo existing = ag.read_text(encoding="utf-8", errors="replace") if ag.exists() else ""
>> "%PYFILE%" echo if "AGENTOPS_AUTOPILOT_MODE" not in existing:
>> "%PYFILE%" echo     ag.write_text(existing.rstrip() + block, encoding="utf-8")
>> "%PYFILE%" echo log.write_text("AgentOps Autopilot installed at " + datetime.datetime.now().isoformat() + "\\n", encoding="utf-8")
>> "%PYFILE%" echo print("OK: AgentOps Autopilot installed")
>> "%PYFILE%" echo print("Created: .opencode/agents/agentops-autopilot.md")
>> "%PYFILE%" echo print("Created: .opencode/commands/autopilot.md")
>> "%PYFILE%" echo print("Created: .opencode/commands/agentmode.md")

call :RUNPY
set "RC=%ERRORLEVEL%"
if "%RC%"=="0" goto SUCCESS

echo.
echo [ERROR] Patch failed.
echo See: %LOG%
pause
exit /b %RC%

:SUCCESS
echo.
echo [OK] Autopilot permission-skip mode added.
echo.
echo Usage in OpenCode:
echo   /agentmode
echo   /autopilot your task here
echo.
echo Or switch primary agent to:
echo   agentops-autopilot
echo.
pause
exit /b 0

:RUNPY
where py >nul 2>nul
if not errorlevel 1 (
  py -3.11 "%PYFILE%"
  if not errorlevel 1 exit /b 0
  py -3 "%PYFILE%"
  if not errorlevel 1 exit /b 0
)

where python >nul 2>nul
if not errorlevel 1 (
  python "%PYFILE%"
  if not errorlevel 1 exit /b 0
)

where python3 >nul 2>nul
if not errorlevel 1 (
  python3 "%PYFILE%"
  if not errorlevel 1 exit /b 0
)

if exist "%LOCALAPPDATA%\Programs\Python\Python311\python.exe" (
  "%LOCALAPPDATA%\Programs\Python\Python311\python.exe" "%PYFILE%"
  if not errorlevel 1 exit /b 0
)

if exist "%USERPROFILE%\AppData\Local\Programs\Python\Python311\python.exe" (
  "%USERPROFILE%\AppData\Local\Programs\Python\Python311\python.exe" "%PYFILE%"
  if not errorlevel 1 exit /b 0
)

if exist "C:\Python311\python.exe" (
  "C:\Python311\python.exe" "%PYFILE%"
  if not errorlevel 1 exit /b 0
)

echo [ERROR] Python failed. Is Python installed?
exit /b 1
