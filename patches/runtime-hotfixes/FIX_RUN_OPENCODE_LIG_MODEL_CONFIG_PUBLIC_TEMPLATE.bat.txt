@python -x "%~f0" %* & pause & exit /b
# -*- coding: utf-8 -*-
"""
FIX_RUN_OPENCODE_LIG_MODEL_CONFIG_PUBLIC_TEMPLATE.bat

Public-repo-safe template for fixing the installed OpenCodeLIG model/provider config.
Do not commit real internal API keys. Closed-network local copies may inject/preserve keys
through USERDATA\secrets\lig-api.env.
"""
from pathlib import Path
import datetime, json, os, shutil, sys

HOME = Path(os.environ.get("USERPROFILE", ""))
ROOT = HOME / "OpenCodeLIG"
WORKSPACE = ROOT / "workspace"
USERDATA = HOME / "OpenCodeLIG_USERDATA"
SECRETS = USERDATA / "secrets"
ENV_FILE = SECRETS / "lig-api.env"
BACKUP_DIR = USERDATA / "backups" / ("model_config_fix_" + datetime.datetime.now().strftime("%Y%m%d_%H%M%S"))

DEFAULT_GATEWAY_BASE_URL = "http://ais.ligdefenseaerospace.com/ai_infra/llm_api"
DEFAULT_MODEL_REF = "lig-qwen-coding/Qwen3.6-27B"

ROUTES = {
    "QWEN": "/gateway/Qwen3.6-27B-vibe_coding_think_off/v1",
    "EXAONE": "/gateway/EXAONE-4.5-33B-vibe_coding_think_off/v1",
    "EXAONE_CHAT": "/gateway/EXAONE-4.5-33B-default_think_off/v1",
    "GEMMA": "/gateway/gemma-4-31B-it/v1",
}

def write_text(path: Path, text: str) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(text, encoding="utf-8", newline="\r\n")

def backup(path: Path) -> None:
    if path.exists():
        BACKUP_DIR.mkdir(parents=True, exist_ok=True)
        safe = str(path).replace(":", "").replace("\\", "_").replace("/", "_")
        shutil.copy2(path, BACKUP_DIR / (safe + ".bak"))

def ensure_env() -> None:
    SECRETS.mkdir(parents=True, exist_ok=True)
    if ENV_FILE.exists():
        print("[OK] env exists:", ENV_FILE)
        return
    text = "\n".join([
        "REM Public template: put real key only on the closed-network PC. Do not commit secrets.",
        f"LIG_GATEWAY_BASE_URL={DEFAULT_GATEWAY_BASE_URL}",
        "LIG_API_KEY=",
        f"LIG_QWEN_BASE_URL={DEFAULT_GATEWAY_BASE_URL}{ROUTES['QWEN']}",
        f"LIG_EXAONE_BASE_URL={DEFAULT_GATEWAY_BASE_URL}{ROUTES['EXAONE']}",
        f"LIG_EXAONE_CHAT_BASE_URL={DEFAULT_GATEWAY_BASE_URL}{ROUTES['EXAONE_CHAT']}",
        f"LIG_GEMMA_BASE_URL={DEFAULT_GATEWAY_BASE_URL}{ROUTES['GEMMA']}",
        "",
    ])
    write_text(ENV_FILE, text)
    print("[WARN] created env template without key:", ENV_FILE)

def config_json() -> dict:
    return {
        "$schema": "https://opencode.ai/config.json",
        "autoupdate": False,
        "share": "disabled",
        "model": DEFAULT_MODEL_REF,
        "provider": {
            "lig-qwen-coding": {
                "npm": "@ai-sdk/openai-compatible",
                "name": "LIG Qwen coding",
                "options": {"baseURL": "{env:LIG_QWEN_BASE_URL}", "apiKey": "{env:LIG_API_KEY}"},
                "models": {"Qwen3.6-27B": {"name": "Qwen3.6 27B coding"}},
            },
            "lig-exaone-coding": {
                "npm": "@ai-sdk/openai-compatible",
                "name": "LIG EXAONE coding",
                "options": {"baseURL": "{env:LIG_EXAONE_BASE_URL}", "apiKey": "{env:LIG_API_KEY}"},
                "models": {"EXAONE-4.5-33B": {"name": "EXAONE 4.5 33B coding"}},
            },
            "lig-exaone-chat": {
                "npm": "@ai-sdk/openai-compatible",
                "name": "LIG EXAONE chat",
                "options": {"baseURL": "{env:LIG_EXAONE_CHAT_BASE_URL}", "apiKey": "{env:LIG_API_KEY}"},
                "models": {"EXAONE-4.5-33B-chat": {"name": "EXAONE 4.5 33B chat"}},
            },
            "lig-gemma-it": {
                "npm": "@ai-sdk/openai-compatible",
                "name": "LIG Gemma IT",
                "options": {"baseURL": "{env:LIG_GEMMA_BASE_URL}", "apiKey": "{env:LIG_API_KEY}"},
                "models": {"gemma-4-31B-it": {"name": "Gemma 4 31B IT"}},
            },
        },
    }

def write_configs() -> None:
    data = json.dumps(config_json(), ensure_ascii=False, indent=2)
    targets = [
        HOME / ".config" / "opencode" / "opencode.json",
        ROOT / "userdata" / "config" / "opencode" / "opencode.json",
        WORKSPACE / "opencode.json",
    ]
    for p in targets:
        backup(p)
        write_text(p, data)
        print("[OK] wrote config:", p)

def write_launcher() -> None:
    run = WORKSPACE / "RUN_OPENCODE_LIG.bat"
    backup(run)
    text = '''@echo off
setlocal EnableExtensions
chcp 65001 >nul
set "PYTHONUTF8=1"
set "PYTHONIOENCODING=utf-8"

set "OCODE_EXE=%USERPROFILE%\OpenCodeLIG\bin\opencode.exe"
set "USERDATA=%USERPROFILE%\OpenCodeLIG_USERDATA"
set "ENV_FILE=%USERDATA%\secrets\lig-api.env"

if not exist "%OCODE_EXE%" (
  echo [ERROR] opencode.exe not found: %OCODE_EXE%
  pause
  exit /b 1
)

if exist "%ENV_FILE%" (
  for /f "usebackq tokens=1,* delims==" %%A in ("%ENV_FILE%") do (
    if not "%%A"=="" if not "%%A:~0,3%"=="REM" set "%%A=%%B"
  )
)

set "OPENCODE_USERDATA=%USERDATA%\opencode_data"
set "XDG_CONFIG_HOME=%USERPROFILE%\OpenCodeLIG\userdata\config"
set "XDG_DATA_HOME=%USERDATA%\opencode_data"
set "XDG_CACHE_HOME=%USERDATA%\opencode_cache"
set "OPENCODE_CONFIG=%USERPROFILE%\OpenCodeLIG\workspace\opencode.json"

cd /d "%USERPROFILE%\OpenCodeLIG\workspace"
"%OCODE_EXE%"
'''
    write_text(run, text)
    print("[OK] wrote launcher:", run)

def main() -> int:
    print("============================================================")
    print("OpenCodeLIG model/provider config repair")
    print("============================================================")
    if not WORKSPACE.exists():
        print("[ERROR] workspace not found:", WORKSPACE)
        return 1
    ensure_env()
    write_configs()
    write_launcher()
    print("[DONE]")
    print("Run:", WORKSPACE / "RUN_OPENCODE_LIG.bat")
    return 0

if __name__ == "__main__":
    raise SystemExit(main())
