@echo off
setlocal EnableExtensions EnableDelayedExpansion
chcp 65001 >nul
set "PYTHONUTF8=1"
set "PYTHONIOENCODING=utf-8"
set "ROOT=%USERPROFILE%\OpenCodeLIG"
set "BIN=%ROOT%\bin"
set "WS=%ROOT%\workspace"
set "USERDATA=%USERPROFILE%\OpenCodeLIG_USERDATA"
set "SEED=%~dp0userdata_seed"
set "SUMS=%~dp0SHA256SUMS.txt"
set "PAYLOAD_EXE=%~dp0payload\opencode.exe"
echo.
echo ============================================================
echo LIG OpenCode patched offline installer
echo ============================================================
echo Target: %ROOT%
echo.
if not exist "%PAYLOAD_EXE%" (
  echo [ERROR] Missing payload\opencode.exe
  pause
  exit /b 1
)
if not exist "%SUMS%" (
  echo [ERROR] Missing SHA256SUMS.txt
  pause
  exit /b 1
)
set "EXPECTED="
for /f "tokens=1,*" %%A in ('findstr /C:"payload/opencode.exe" "%SUMS%"') do set "EXPECTED=%%A"
if not defined EXPECTED (
  echo [ERROR] payload/opencode.exe entry missing in SHA256SUMS.txt
  pause
  exit /b 1
)
set "ACTUAL="
for /f "tokens=1" %%A in ('certutil -hashfile "%PAYLOAD_EXE%" SHA256 ^| findstr /R "^[0-9A-Fa-f][0-9A-Fa-f]*$"') do if not defined ACTUAL set "ACTUAL=%%A"
if not defined ACTUAL (
  echo [ERROR] Failed to calculate payload checksum
  pause
  exit /b 1
)
if /I not "!EXPECTED!"=="!ACTUAL!" (
  echo [ERROR] payload/opencode.exe checksum mismatch
  echo Expected: !EXPECTED!
  echo Actual:   !ACTUAL!
  pause
  exit /b 1
)
if exist "%ROOT%" (
  set "BACKUP=%USERPROFILE%\OpenCodeLIG_backup_%RANDOM%"
  echo Existing installation found. Backing up to !BACKUP!
  robocopy "%ROOT%" "!BACKUP!" /E /NFL /NDL /NJH /NJS /NP >nul
  if errorlevel 8 (
    echo [ERROR] Existing installation backup failed
    pause
    exit /b 1
  )
)
if not exist "%BIN%" mkdir "%BIN%"
if not exist "%WS%" mkdir "%WS%"
copy /y "%PAYLOAD_EXE%" "%BIN%\opencode.exe" >nul
if errorlevel 1 (
  echo [ERROR] Failed to copy opencode.exe
  pause
  exit /b 1
)
rem /PURGE mirrors the workspace to exactly match this package so stale files
rem from a previous install (runtime output folders, results, snapshots)
rem never linger. /XD must name SOURCE-side paths (robocopy matches the dir
rem being scanned, not the destination) so tools\ is neither re-copied from
rem the package nor purged from an existing install - keeping user-imported
rem offline binaries (Obsidian/OCR/wheels) intact across reinstalls.
robocopy "%~dp0workspace" "%WS%" /E /PURGE /XD "%~dp0workspace\tools" "%~dp0workspace\tests" "%~dp0workspace\docs\archive" /NFL /NDL /NJH /NJS /NP >nul
if errorlevel 8 (
  echo [ERROR] Workspace copy failed
  pause
  exit /b 1
)
rem Ensure tools\ exists (its README ships in the package but /XD skipped it).
if not exist "%WS%\tools" mkdir "%WS%\tools"
rem The package's own wheelhouse is not user data (unlike Obsidian/OCR the
rem user brings in later), so copy it into the installed tree too: this is
rem what makes launch\install-tools.bat retryable from the installed
rem workspace instead of only from the install package.
if exist "%~dp0workspace\tools\wheelhouse" (
  robocopy "%~dp0workspace\tools\wheelhouse" "%WS%\tools\wheelhouse" /E /NFL /NDL /NJH /NJS /NP >nul
)

rem Expose `ocd` as a global command: copy the launcher into bin and add bin
rem to the USER PATH safely (PowerShell SetEnvironmentVariable - no setx 1024-char
rem truncation, idempotent). New terminal picks it up: `cd project` then `ocd`.
if exist "%WS%\launch\ocd.bat" (
  copy /y "%WS%\launch\ocd.bat" "%BIN%\ocd.bat" >nul
  powershell -NoProfile -Command "$b=[Environment]::GetEnvironmentVariable('Path','User'); if($null -eq $b){$b=''}; if($b -notlike '*OpenCodeLIG\bin*'){[Environment]::SetEnvironmentVariable('Path',($b.TrimEnd(';')+';%BIN%'),'User'); Write-Host '[OK] ocd command registered on PATH (open a NEW terminal to use it).'} else {Write-Host '[OK] ocd already on PATH.'}" 2>nul || echo [WARN] Could not auto-register ocd on PATH; add %BIN% to PATH manually.
)
rem Seed OpenCodeLIG_USERDATA with the pre-filled gateway secret so the
rem assistant connects with zero configuration. Non-destructive: only seeds
rem the secret when it does not already exist (USERDATA is inviolable - an
rem existing key/memory/wiki is never overwritten here).
if not exist "%USERDATA%\secrets" mkdir "%USERDATA%\secrets"
if exist "%SEED%" (
  if not exist "%USERDATA%\secrets\lig-api.env" (
    robocopy "%SEED%" "%USERDATA%" /E /NFL /NDL /NJH /NJS /NP >nul
    if errorlevel 8 (
      echo [WARN] USERDATA seed copy reported errors; check %USERDATA%
    ) else (
      echo [OK] Gateway secret seeded: %USERDATA%\secrets\lig-api.env
    )
  ) else (
    echo [SKIP] Existing USERDATA secret kept: %USERDATA%\secrets\lig-api.env
  )
)

rem opencode.json(사내 EXAONE/Qwen provider + autoupdate/share/mdns off)은 패키지에
rem 동봉되어 robocopy 로 이미 복사됨 — 여기서 최소본으로 덮어쓰지 않는다(그러면 기본
rem 모델로 열려 '빅 피클' 문제 재발). 만약 없으면(구 패키지) 안내만.
if not exist "%WS%\opencode.json" echo [WARN] opencode.json 이 없습니다 - 모델 설정 확인 필요.

rem Remove stale old-mode files from a previous install (enforce primary=agent).
rem Best-effort: skip silently if Python is unavailable (launcher retries at start).
pushd "%WS%"
py -3.11 -m agent_ops.clean_stale 2>nul || python -m agent_ops.clean_stale 2>nul
popd

if exist "%WS%\RUN_OPENCODE_LIG.bat" (
  echo [OK] Unified launcher ready: %WS%\RUN_OPENCODE_LIG.bat
) else (
  echo [ERROR] RUN_OPENCODE_LIG.bat missing after workspace copy
  pause
  exit /b 1
)
(
  echo @echo off
  echo "%%USERPROFILE%%\OpenCodeLIG\bin\opencode.exe" --version
  echo if errorlevel 1 exit /b 1
  echo echo [OK] patched opencode.exe exists and runs.
) > "%WS%\VERIFY_OFFLINE_INSTALL.bat"
echo.
rem Offline dependency install (bundled wheels) so the pilot needs no extra
rem step: openpyxl/python-docx/python-pptx (real Office files), markitdown
rem (read docs), rapidocr (screen OCR). Best-effort: never fails the install.
set "WH=%~dp0workspace\tools\wheelhouse"
if exist "%WH%" (
  echo Installing bundled offline dependencies ^(one-time^)...
  pushd "%WS%"
  py -3.11 -m pip install --no-index --find-links "%WH%" openpyxl python-docx python-pptx "markitdown[pdf,docx,pptx,xlsx]" rapidocr-onnxruntime 1>nul 2>nul || python -m pip install --no-index --find-links "%WH%" openpyxl python-docx python-pptx "markitdown[pdf,docx,pptx,xlsx]" rapidocr-onnxruntime 1>nul 2>nul
  if errorlevel 1 (
    echo [WARN] Some offline deps did not install ^(optional features^). Retry later: launch\install-tools.bat
  ) else (
    echo [OK] Offline dependencies installed.
  )
  popd
)
echo [OK] Installed.
echo Run:
echo   %WS%\RUN_OPENCODE_LIG.bat
echo Verify:
echo   %WS%\VERIFY_OFFLINE_INSTALL.bat
echo Pending check:
echo   %WS%\점검용_전체확인.bat
echo Docs:
echo   %WS%\docs\사용법\GUIDE.md
echo.
pause
exit /b 0
